MYSQL CODE

ACCOUNT:
create table ACCOUNT( 
ACC_USERNAME VARCHAR(100) NOT NULL PRIMARY KEY, 
ACC_FNAME VARCHAR(25) NOT NULL, 
ACC_LNAME VARCHAR(25) NOT NULL, 
ACC_SUFIX VARCHAR(20), 
ACC_MIDDLE VARCHAR(75), 
ACC_MAIDEN VARCHAR(75), 
ACC_PREFIX VARCHAR(20), 
ACC_HONNARY VARCHAR(50), 
ACC_DOB DATE, 
ACC_SCHOOLID VARCHAR(20) NOT NULL,
ACC_SCHOOLNAME VARCHAR(150) NOT NULL);


EMAIL:
CREATE TABLE EMAIL( 
EMA_ADDRESS VARCHAR(100) PRIMARY KEY, 
ACC_USERNAME VARCHAR(100) NOT NULL, 
FOREIGN KEY (ACC_USERNAME) REFERENCES ACCOUNT(ACC_USERNAME));


INSTRUCTOR:
CREATE TABLE INSTRUCTOR( 
ACC_USERNAME VARCHAR(100) PRIMARY KEY, 
ACC_FNAME VARCHAR(25) NOT NULL, 
ACC_LNAME VARCHAR(25) NOT NULL, 
FOREIGN KEY (ACC_USERNAME) REFERENCES ACCOUNT(ACC_USERNAME));


STUDENT:
CREATE TABLE STUDENT(
ACC_USERNAME VARCHAR(100) PRIMARY KEY, 
ACC_FNAME VARCHAR(25) NOT NULL, 
ACC_LNAME VARCHAR(25) NOT NULL, 
FOREIGN KEY (ACC_USERNAME) REFERENCES ACCOUNT(ACC_USERNAME));


CLASS:
CREATE TABLE CLASS( 
CLS_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
CLS_NAME VARCHAR(100) NOT NULL, 
ACC_USERNAME VARCHAR(100) NOT NULL, 
ACC_FNAME VARCHAR(25) NOT NULL, 
ACC_LNAME VARCHAR(25) NOT NULL, 
CLS_SDATE DATE NOT NULL, 
CLS_EDATE DATE NOT NULL,
FOREIGN KEY ACC_USERNAME REFERENCES INSTRUCTOR(ACC_USERNAME));


COURSE:
CREATE TABLE COURSE( 
CRS_ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
CLS_ID INT NOT NULL, 
CLS_SDATE DATE NOT NULL, 
CLS_EDATE DATE NOT NULL,
FOREIGN KEY CLS_ID REFERENCES CLASS(CLS_ID));


REQUIREMENT:
CREATE TABLE REQUIREMENT( 
REQ_ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
CLS_ID INT NOT NULL, 
CRS_ID INT NOT NULL, 
REQ_REQUIRED BOOLEAN NOT NULL,
FOREIGN KEY CLS_ID REFERENCES CLASS(CLS_ID),
FOREIGN KEY CRS_ID REFERENCES COURSE(CRS_ID));


GRADE:
CREATE TABLE GRADE( 
GRD_CODE CHAR(5) NOT NULL PRIMARY KEY, 
GRD_DESCRIPTION VARCHAR(50) NOT NULL, 
GRD_VALUE INT);


COMPLETION:
CREATE TABLE COMPLETION( 
COM_ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
ACC_USERNAME VARCHAR(100) NOT NULL, 
GRD_GRADE CHAR(5),
FOREIGN KEY ACC_USERNAME REFERENCES ACCOUNT(ACC_USERNAME),
FOREIGN KEY GRD_GRADE REFERENCES GRADE(GRD_GRADE));
